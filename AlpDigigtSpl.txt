#include <stdio.h>
#include <ctype.h>

int main() {
    char character;

    printf("Enter a character: ");
    scanf("%c", &character);

    if (isalpha(character)) {
        printf("%c is an alphabet.\n", character);
    } else if (isdigit(character)) {
        printf("%c is a digit.\n", character);
    } else {
        printf("%c is a special symbol.\n", character);
    }

    return 0;
}
